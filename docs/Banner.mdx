## Display

To take full width of the parent element, use ***width*** prop.

```jsx
const BannerDemo: React.FC = () => {
  const wrapperRef = useRef();
  const [ width, setWidth ] = useState(0);

  useEffect(() => {
    window.addEventListener('resize', handleResize);
    return() => window.removeEventListener('resize', handleResize);
  })

  const handleResize = () => {
    wrapperRef.current && setWidth(wrapperRef.current.clientWidth);
  };

  useEffect(() => {
    wrapperRef.current && setWidth(wrapperRef.current.clientWidth);
  }, [wrapperRef]);

  return(
    <div style={{ width: '100%' }} ref={wrapperRef}>
      {wrapperRef.current && (
        <Banner width={`${width}px`}>
          {items.map((item, index) => 
            <BannerItem key={index} item={item} />)}
        </Banner>)}
    </div>
  );
};
```
